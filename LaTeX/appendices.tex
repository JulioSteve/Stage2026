\section{Representations of quantum mechanics}\label{representations}
\subsection*{Brief complement of quantum mechanics}
An average is defined in quantum mechanics by
\begin{align*}
    \expval{O}_\psi&=\expval{O}{\psi},\\
    \expval{O}_\rho&=\Tr{\rho O},
\end{align*}
where the subscripts $\ket{\psi}$ and $\rho$ refer respectively to the representation of pure states or
to density matrix more general formalism. 
\subsection{Schrödinger and Heisenberg pictures}\noindent
In this appendix we denote the pictures with their acronyms and lighten the notation of the total state $\rho_T\equiv\rho$. However, this could be misleading because $S$ in this appendix stands for \textit{``in the Schrödinger picture''}, whereas, $S$ in the main chapters means \textit{``state of the system S''}. Nevertheless, we only consider the state in a closed quantum system in the following so we cannot encounter that kind of situation.

In the Schrödinger picture we saw that $\ket{\psi(t)}=U(t)\ket{\psi}$, where $U(t)=e^{-\frac{i}{\hbar}Ht}$ in the case of a time-independent Hamiltonian with $t_0=0$. Nevertheless, the following results can be derived in the general case of a time dependent Hamiltonian,~\cite{Breuer}. Considering the average value of an observable $A_S(t)$ in the Schrödinger picture, we find
\begin{align}
    \expval{A_S(t)} &= \expval{A_S}{\psi(t)}\nonumber\\
    &= \expval{\underbrace{U^\dagger(t)A_S(t)U(t)}_{A_H(t)}}{\psi}.
\end{align}
This calculation naturally reveals the Heisenberg representation of observables and states. Indeed, we have that $\forall t,\;\ket{\psi(t)}_H=\ket{\psi(t_0)}_S\equiv\ket{\psi}_H$ which links both pictures at the initial time. Furthermore, we can link both representation by calculating the evolution of an observable in the Heisenberg picture:
\begin{equation}
    \boxed{\dert{A_H} = \frac{i}{\hbar}\left[H_{H},A_{H}\right] + \derpart{A_S}\label{eq:HSlink}}.
\end{equation}
This link can even lead to the famous \textit{Ehrenfest theorem} for quantum mechanics:
\begin{equation}
    \boxed{\dert{\expval{A}}=\frac{i}{\hbar}\expval{\qty[H(t),A(t)]}+\expval{\derpart{A}}}.
\end{equation}
We observe that if the observable doesn't depend explicitly on time (thus constant), then the dynamics of the operator are fully described by the Heisenberg Eq.~(\ref{eq:HSlink}).

\subsection{Interaction picture}\noindent
As mentioned in~\ref{OQS}, we use perturbation theory to describe the interaction between the system $S$ and the environment $E$,
\begin{align}
    H_T &= \overbrace{H_S\otimes\mathbb{I}_E+\mathbb{I}_S\otimes H_E}^{H_0}+H_I\tag{\ref{eq:Htotal}},\\
    U(t,t_0) &= U_0(t,t_0)U_I(t,t_0)\tag{\ref{eq:Utotal}}.
\end{align}
Because $\ket{\psi(t)}_S=U(t)\ket{\psi}_S$, we can write that
\begin{equation}\label{eq:evolrhoU}
    \rho_S(t) = U(t)\rho_{S}U^\dagger(t).
\end{equation}
If we now try to calculate the average of an operator (in the total space $\mathcal{H}_T$) using the Schrödinger picture, we can find a new interesting relation by injecting Eq.~(\ref{eq:evolrhoU},\ref{eq:Utotal}) in the average:
\begin{align*}
    \expval{O_S(t)} &= \Tr{\rho_{S}(t)O_S(t)}\\
    &= \mathrm{Tr}\,[\underbrace{U_I(t)\rho_{S}U_I^\dagger(t)}_{\tilde{\rho}(t)}\,\underbrace{U_0^\dagger(t)O_S(t)U_0(t)}_{\tilde{O}(t)}].
\end{align*}
We can thus define in the interaction picture:
\begin{align}
    \tilde{\rho}(t)&=U_I(t)\rho_{S}U_I^\dagger(t)\tag{\ref{eq:rhoIP}},\\
    \tilde{O}(t) &= U_0^\dagger(t)O_S(t)U_0(t)\tag{\ref{eq:OIP}}.
\end{align}
We can now reverse Eq.~(\ref{eq:evolrhoU}) and then use Eq.~(\ref{eq:Utotal}):
\begin{equation*}
    \rho_S = U^\dagger(t)\rho_S(t)U(t) = U_I^\dagger(t)U^\dagger_0(t)\rho_S(t)U_0(t)U_I(t),
\end{equation*}
and, by injecting this into Eq.~(\ref{eq:rhoIP}), we get
\begin{align}
    \tilde{\rho}(t) &= U_I(t)U_I^\dagger(t)U^\dagger_0(t)\rho_S(t)U_0(t)U_I(t)U^\dagger_I(t)\nonumber\\
    &= U^\dagger_0(t)\rho_S(t)U_0(t)\tag{\ref{eq:rhoIP}}.
\end{align}
So that we find the results used in the main part of this work
\begin{equation}
    \boxed{\tilde{\rho}(t)=U_I(t)\rho_{S}U_I^\dagger(t)= U^\dagger_0(t)\rho_S(t)U_0(t)}.
\end{equation}
